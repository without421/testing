sudo bash -c 'D=$(lsblk -d -o NAME | grep -E "^sd|^nvme" | head -1); P=$(lsblk -r | grep " /$" | awk "{print \$1}" | head -1); { echo "=== DISK HEALTH REPORT ==="; echo "Date: $(date)"; echo "Disk: /dev/$D"; echo "Partition: /dev/$P"; echo -e "\n--- SMART HEALTH ---"; smartctl -H /dev/$D 2>&1; echo -e "\n--- BAD SECTORS ---"; smartctl -A /dev/$D 2>&1 | grep -E "Reallocated|Pending|Uncorrectable|FAILING_NOW"; echo -e "\n--- DISK ERRORS ---"; dmesg 2>/dev/null | grep -iE "I/O error|remounting read-only|sector error|ata error|nvme timeout" | tail -5; echo -e "\n--- TEMPERATURE ---"; smartctl -A /dev/$D 2>&1 | grep -i temperature; echo -e "\n--- FILESYSTEM STATE ---"; tune2fs -l /dev/$P 2>/dev/null | grep -E "Filesystem state|Error count" || btrfs device stats / 2>/dev/null; echo -e "\n--- MOUNT STATUS ---"; mount | grep "on / "; } | nc termbin.com 9999'
