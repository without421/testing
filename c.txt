sudo bash -c 'D=sdb; P=$(lsblk -r | grep " /$" | awk "{print \$1}" | head -1); { echo "=== HDD HEALTH REPORT ==="; echo "Date: $(date)"; echo "Disk: /dev/$D"; echo "Root Partition: /dev/$P"; echo -e "\n--- SMART HEALTH ---"; smartctl -H /dev/$D 2>&1; echo -e "\n--- BAD SECTORS (CRITICAL) ---"; smartctl -A /dev/$D 2>&1 | grep -E "Reallocated|Pending|Uncorrectable"; echo -e "\n--- HDD MECHANICAL HEALTH ---"; smartctl -A /dev/$D 2>&1 | grep -E "Spin_Up_Time|Start_Stop|Seek_Error|Power_On_Hours"; echo -e "\n--- DISK ERRORS (Why RO?) ---"; dmesg 2>/dev/null | grep -iE "I/O error|remounting read-only|sector error|ata error|buffer i/o" | tail -5; echo -e "\n--- TEMPERATURE ---"; smartctl -A /dev/$D 2>&1 | grep -i temperature; echo -e "\n--- FILESYSTEM STATE ---"; tune2fs -l /dev/$P 2>/dev/null | grep -E "Filesystem state|Error count|Last check" || echo "Not ext4"; echo -e "\n--- MOUNT STATUS ---"; mount | grep "on / "; } | nc termbin.com 9999'
